38 double SourceTermCoefficient::Eval(mfem::ElementTransformation &T,
39  const mfem::IntegrationPoint &ip) {
40  switch (SourceTerm::from(this->source_term_name_)) {
41  case SourceTerm::Null: {
42  const auto src_value = 0.;
43  return src_value;
44  }
45 
46  case SourceTerm::Sinusoide2D: {
47  const auto time = this->GetTime();
48  const auto u = std::exp(-2. * time) * std::sin(ip.x + ip.y);
49  const auto src_value = u * u * u - u;
50  return -1. * src_value / (0.09);
51  }
52  default:
53  throw std::runtime_error(
54  "SourceTermCoefficient::Eval: only Null, Sinusoide2D source term are available");
55  break;
56  }
57 }
