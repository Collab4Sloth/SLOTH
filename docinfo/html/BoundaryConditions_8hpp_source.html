50 template <class T, int DIM>
51 template <class... Args>
52 BoundaryConditions<T, DIM>::BoundaryConditions(SpatialDiscretization<T, DIM> *spatial,
53  const Args &...boundaries) {
54  this->fespace_ = spatial->get_finite_element_space();
55  const auto &mesh_max_bdr_attributes = spatial->get_max_bdr_attributes();
56 
57  auto bdrs = std::vector<Boundary>{boundaries...};
58 
59  Dirichlet_bdr_.SetSize(mesh_max_bdr_attributes);
60  Dirichlet_value_.SetSize(mesh_max_bdr_attributes);
61  bool exist_periodic_bdr = {false};
62  [bdrs, &exist_periodic_bdr]() {
63  bool is_periodic = false;
64  for (const auto &bdr : bdrs) {
65  if (bdr.is_essential_boundary()) {
66  is_periodic = true;
67  break;
68  }
69  }
70  return is_periodic;
71  };
72  bool test_periodic_bdr = (mesh_max_bdr_attributes != bdrs.size()) && exist_periodic_bdr;
73  bool test_standard_bdr = mesh_max_bdr_attributes == bdrs.size();
74  if (test_periodic_bdr || test_standard_bdr) {
75  for (const auto &bdr : bdrs) {
76  const auto &id = bdr.get_boundary_index();
77  if (bdr.is_essential_boundary()) {
78  Dirichlet_bdr_[id] = 1;
79  } else {
80  Dirichlet_bdr_[id] = 0;
81  }
82  Dirichlet_value_[id] = bdr.get_boundary_value();
83  }
84  this->fespace_->GetEssentialTrueDofs(this->Dirichlet_bdr_, this->ess_tdof_list_);
85  } else {
86  throw std::runtime_error(
87  "BoundaryConditions::BoundaryConditions(): user-defined boundaries " +
88  std::to_string(bdrs.size()) +
89  " are unconsistent with the total number of boundaries associated to the mesh " +
90  std::to_string(mesh_max_bdr_attributes));
91  }
92 }

109 template <class T, int DIM>
110 void BoundaryConditions<T, DIM>::SetBoundaryConditions(mfem::Vector &u) {
111  mfem::Array<int> tmp_array_bdr(this->Dirichlet_bdr_.Size());
112  for (auto i = 0; i < this->Dirichlet_bdr_.Size(); i++) {
113  tmp_array_bdr = 0;
114  mfem::Array<int> dof;
115  if (this->Dirichlet_bdr_[i] > 0) {
116  tmp_array_bdr[i] = 1;
117  this->fespace_->GetEssentialTrueDofs(tmp_array_bdr, dof);
118  u.SetSubVector(dof, this->Dirichlet_value_[i]);
119  }
120  }
121 } // end of SetBoundaryConditions
