209 template <class T, int DIM, class NLFI>
210 void PhaseFieldOperatorBase<T, DIM, NLFI>::SetTransientParameters(const double dt,
211  const mfem::Vector &u) {
212  delete N;
213  delete reduced_oper;
214 
216  // PhaseField reduced operator N
218  N = new mfem::NonlinearForm(this->fespace_);
219  mfem::GridFunction un_gf(this->fespace_);
220  un_gf.SetFromTrueDofs(u);
221  mfem::GridFunction un(this->fespace_);
222  un.SetFromTrueDofs(u);
223 
224  NLFI *nlfi_ptr = set_nlfi_ptr(dt, u);
225  N->AddDomainIntegrator(nlfi_ptr);
226  N->SetEssentialTrueDofs(this->ess_tdof_list_);
227 
228  reduced_oper = new PhaseFieldReducedOperator(M, N);
229 
231  // Newton Solver
233  this->ut_solver_.SetSolverParameters(
234  this->newton_solver_, NewtonDefaultConstant::print_level,
235  NewtonDefaultConstant::iterative_mode, NewtonDefaultConstant::iter_max,
236  NewtonDefaultConstant::rel_tol, NewtonDefaultConstant::abs_tol);
237  // TODO(ci230846) : cette partie devra etre generalisee pour un solveur iteratif
238  J_solver = new mfem::UMFPackSolver;
239  this->ut_solver_.BuildSolver(this->newton_solver_, *J_solver, *reduced_oper);
240 }

275 template <class T, int DIM, class NLFI>
276 void PhaseFieldOperatorBase<T, DIM, NLFI>::Mult(const mfem::Vector &u, mfem::Vector &du_dt) const {
277  const auto sc = height;
278  mfem::Vector v(u.GetData(), sc);
279  mfem::Vector dv_dt(du_dt.GetData(), sc);
280 
281  N->Mult(v, z);
282 
283  // Source term
284  if (!this->source_term_name_.empty()) {
285  mfem::Vector source_term;
286  this->get_source_term(source_term);
287  z -= source_term;
288  }
289 
290  z.Neg(); // z = -z
291  M_solver.Mult(z, dv_dt);
292 }

301 template <class T, int DIM, class NLFI>
302 void PhaseFieldOperatorBase<T, DIM, NLFI>::ImplicitSolve(const double dt, const mfem::Vector &u,
303  mfem::Vector &du_dt) {
304  const auto sc = height;
305  mfem::Vector v(u.GetData(), sc);
306  mfem::Vector dv_dt(du_dt.GetData(), sc);
307  // // Solve the equation:
308  // // du_dt = M^{-1}*[-K(u + dt*du_dt)]
309  // // for du_dt
310 
311  this->bcs_->SetBoundaryConditions(v);
312  this->SetTransientParameters(dt, v);
313 
314  reduced_oper->SetParameters(dt, &v);
315 
316  // Source term
317  mfem::Vector source_term;
318  if (!this->source_term_name_.empty()) {
319  this->get_source_term(source_term);
320  }
321 
322  dv_dt = v;
323  dv_dt *= (1. / dt);
324  this->newton_solver_.Mult(source_term, dv_dt);
325  dv_dt.SetSubVector(this->ess_tdof_list_, 0.0); // pour Dirichlet ... uniquement?
326  // std::cout << " PhaseFieldOperatorBase this->newton_solver_->Mult " << std::endl;
327 
328  MFEM_VERIFY(this->newton_solver_.GetConverged(), "Nonlinear solver did not converge.");
329 }

337 template <class T, int DIM, class NLFI>
338 void PhaseFieldOperatorBase<T, DIM, NLFI>::ComputeEnergies(
339  const std::tuple<int, double, double> &iter, const mfem::Vector &u) {
340  mfem::GridFunction un_gf(this->fespace_);
341  un_gf.SetFromTrueDofs(u);
342  mfem::GridFunction gf(this->fespace_);
343  EnergyCoefficient g(&un_gf, 0.5 * this->lambda_, this->omega_);
344  gf.ProjectCoefficient(g);
345  mfem::GridFunction sigf(this->fespace_);
346  EnergyCoefficient sig(&un_gf, this->lambda_, 0.);
347  sigf.ProjectCoefficient(sig);
348 
349  // Calcul de l'intÃ©grale de l'objet FunctionCoefficient sur le domaine
350  mfem::ConstantCoefficient zero(0.);
351  const auto energy = gf.ComputeL1Error(zero);
352  this->energy_density_.try_emplace(iter, energy);
353  const auto interfacial_energy = sigf.ComputeL1Error(zero);
354  this->energy_interface_.try_emplace(iter, interfacial_energy);
355 }
