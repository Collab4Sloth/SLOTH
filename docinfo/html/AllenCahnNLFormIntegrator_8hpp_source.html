77 template <ThermodynamicsPotentialDiscretization SCHEME, ThermodynamicsPotentials ENERGY,
78  Mobility MOBI>
79 void AllenCahnNLFormIntegrator<SCHEME, ENERGY, MOBI>::AssembleElementVector(
80  const mfem::FiniteElement& el, mfem::ElementTransformation& Tr, const mfem::Vector& elfun,
81  mfem::Vector& elvect) {
82  int nd = el.GetDof();
83  int dim = el.GetDim();
84  int spaceDim = Tr.GetSpaceDim();
85  dshape.SetSize(nd, dim);
86  shape.SetSize(nd);
87  invdfdx.SetSize(dim, spaceDim);
88  vec.SetSize(dim);
89  pointflux.SetSize(spaceDim);
90 
91  elvect.SetSize(nd);
92  const mfem::IntegrationRule* ir =
93  &mfem::IntRules.Get(el.GetGeomType(), 2 * el.GetOrder() + Tr.OrderW());
94  elvect = 0.0;
95  for (int i = 0; i < ir->GetNPoints(); i++) {
96  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
97  el.CalcDShape(ip, dshape); // dphi
98  el.CalcShape(ip, shape); // phi
99  Tr.SetIntPoint(&ip);
100 
101  const auto u = elfun * shape;
102  const auto un = u_old.GetValue(Tr, ip);
103 
104  const auto W = this->energy_first_derivative_potential_.getPotentialFunction(un);
105  const auto Wprime = W(u);
106  const auto MOB = this->mobility_function_.getMobilityFunction(un);
107  const auto Mphi = MOB(this->mob_);
108 
109  CalcAdjugate(Tr.Jacobian(), invdfdx); // invdfdx = adj(J)
110 
111  dshape.MultTranspose(elfun, vec);
112  invdfdx.MultTranspose(vec, pointflux);
113 
114  // Energy contribution + source
115  const auto energy = this->omega * Wprime;
116  const auto fun_val = Mphi * energy;
117 
118  // Given phi, compute (w'(phi)-f, v), v is shape function
119  const double ww = ip.weight * Tr.Weight() * fun_val;
120  add(elvect, ww, shape, elvect);
121 
122  // Laplacian : given u, compute (grad(u), grad(v)), v is shape function.
123  double w;
124  w = Mphi * ip.weight * this->lambda / Tr.Weight();
125  pointflux *= w;
126  invdfdx.Mult(pointflux, vec);
127  dshape.AddMult(vec, elvect);
128  }
129 }

143 void AllenCahnNLFormIntegrator<SCHEME, ENERGY, MOBI>::AssembleElementGrad(
144  const mfem::FiniteElement& el, mfem::ElementTransformation& Tr, const mfem::Vector& elfun,
145  mfem::DenseMatrix& elmat) {
146  int nd = el.GetDof();
147  int dim = el.GetDim();
148  int spaceDim = Tr.GetSpaceDim();
149  bool square = (dim == spaceDim);
150  double w;
151 
152  shape.SetSize(nd);
153  dshape.SetSize(nd, dim);
154  dshapedxt.SetSize(nd, spaceDim);
155  elmat.SetSize(nd);
156 
157  const mfem::IntegrationRule* ir =
158  &mfem::IntRules.Get(el.GetGeomType(), 2 * el.GetOrder() + Tr.OrderW());
159 
160  elmat = 0.0;
161  for (int i = 0; i < ir->GetNPoints(); i++) {
162  const mfem::IntegrationPoint& ip = ir->IntPoint(i);
163  el.CalcDShape(ip, dshape); // dphi
164  const auto u = elfun * shape;
165  const auto un = u_old.GetValue(Tr, ip);
166 
167  const auto W = this->energy_second_derivative_potential_.getPotentialFunction(un);
168  const auto Wsecond = W(u);
169  const auto MOB = this->mobility_function_.getMobilityFunction(un);
170  const auto Mphi = MOB(this->mob_);
171 
172  Tr.SetIntPoint(&ip);
173  w = Tr.Weight(); // det(J)
174  // std::cout << " SQUARE ? " << square << std::endl;
175  w = ip.weight / (square ? w : w * w * w);
176  // AdjugateJacobian = / adj(J), if J is square
177  // \ adj(J^t.J).J^t, otherwise
178 
179  // Tr.AdjugateJacobian() det(J)J-1
180 
181  w *= Mphi * this->lambda;
182 
183  // dshapedxt = det(J)J-1 dshape
184  Mult(dshape, Tr.AdjugateJacobian(), dshapedxt);
185  // elmat += w * dshapedxt * dshapedxt^T
186  AddMult_a_AAt(w, dshapedxt, elmat);
187 
188  // (this->omega * secondDerivativedoubleWellPotential(elfun * shape) +
189  // this->alpha * secondDerivativeInterpolationPotential(elfun * shape)) *
190  // Compute w'(u)*(du,v), v is shape function
191  double fun_val = (Mphi * this->omega * Wsecond) * ip.weight * Tr.Weight(); // w'(u)
192  // elmat += fun_val * shape * shape^T
193  AddMult_a_VVt(fun_val, shape, elmat); // w'(u)*(du, v)
194  }
195 }
