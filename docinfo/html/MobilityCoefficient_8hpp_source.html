58 template <Mobility MOBI>
59 double MobilityCoefficient<MOBI>::Eval(mfem::ElementTransformation &T,
60  const mfem::IntegrationPoint &ip) {
61  switch (MOBI) {
62  case Mobility::Constant:
63  return this->mobility_;
64  case Mobility::Degenerated:
65  const auto xx = gf.GetValue(T.ElementNo, ip);
66  const auto a_xx = std::max(std::min(xx, 1.), 0.);
67  const auto degenerated_mob_func = this->mobility_ *
68  std::pow(1. - a_xx, this->degenerate_order_) *
69  std::pow(a_xx, this->degenerate_order_);
70  return degenerated_mob_func;
71  default:
72  throw std::runtime_error(
73  "MobilityCoefficient::Eval: only constant and degenerated mobilities are available");
74  break;
75  }
76 }
