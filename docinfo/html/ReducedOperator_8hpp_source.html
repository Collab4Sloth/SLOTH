68 mfem::Operator &PhaseFieldReducedOperator::GetGradient(const mfem::Vector &k) const {
69  // std::cout << " PhaseFieldReducedOperator::GetGradient " << std::endl;
70  delete Jacobian;
71  Jacobian = Add(1.0, M->SpMat(), 0.0, M->SpMat());
72  add(*unk, dt, k, z);
73  mfem::SparseMatrix *grad_N = dynamic_cast<mfem::SparseMatrix *>(&N->GetGradient(z));
74  Jacobian->Add(dt, *grad_N);
75  return *Jacobian;
76 }
