53 Boundary::Boundary(const std::string &boundary_name, const int &boundary_index,
54  const std::string &boundary_type)
55  : boundary_name_(boundary_name), boundary_index_(boundary_index) {
56  switch (BoundaryConditionType::from(boundary_type)) {
57  case BoundaryConditionType::Dirichlet:
58  this->is_essential_boundary_ = true;
59  this->is_periodic_boundary_ = false;
60  break;
61  case BoundaryConditionType::Neumann:
62  case BoundaryConditionType::Robin:
63  this->is_essential_boundary_ = false;
64  this->is_periodic_boundary_ = false;
65  break;
66  case BoundaryConditionType::Periodic:
67  this->is_periodic_boundary_ = true;
68  this->is_essential_boundary_ = false;
69  break;
70  default:
71  throw std::runtime_error(
72  "Boundary::Boundary(): only Dirichlet, Neumann, Periodic and Robin BoundaryConditionType "
73  "are available");
74  break;
75  }
76 }
