827  ThermodynamicsPotentials POTENTIAL>

829 std::function<double(const double&)>
830 PotentialFunctions<ORDER, SCHEME, POTENTIAL>::getPotentialFunction(Args... args) {
831  switch (POTENTIAL) {
832  case ThermodynamicsPotentials::W:
833  return this->getW(args...);
834  case ThermodynamicsPotentials::F:
835  return this->getF(args...);
836  case ThermodynamicsPotentials::H:
837  return this->getH(args...);
838  case ThermodynamicsPotentials::X:
839  return this->getX(args...);
840  default:
841  throw std::runtime_error(
842  "PotentialFunctions::getPotentialFunctions: double well, H interpolation and identity "
843  "potential function are available");
844  break;
845  }
846 }
