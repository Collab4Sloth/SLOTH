105 template <class PST, class OPE, class VAR>
106 void TimeDiscretization<PST, OPE, VAR>::set_ODE_solver(const std::string& ode_solver) {
107  switch (TimeScheme::from(ode_solver)) {
108  case TimeScheme::EulerExplicit: {
109  this->ode_solver_ = new mfem::ForwardEulerSolver;
110  break;
111  }
112  case TimeScheme::EulerImplicit: {
113  this->ode_solver_ = new mfem::BackwardEulerSolver;
114  break;
115  }
116  case TimeScheme::RungeKutta4: {
117  this->ode_solver_ = new mfem::SDIRK33Solver;
118  break;
119  }
120  default:
121  throw std::runtime_error(
122  "TimeDiscretization::set_ODE_solver: EulerImplicit, EulerExplicit, Rungekutta4 are "
123  "available");
124  break;
125  }
126 }

198 template <class PST, class OPE, class VAR>
199 double TimeDiscretization<PST, OPE, VAR>::compute_real_time_step(const double& current_time) {
200  auto dt_real = this->time_step_;
201  if ((current_time + this->time_step_) + 0.5 * this->time_step_ > this->final_time_) {
202  this->last_step_ = true;
203  dt_real = this->final_time_ - current_time;
204  }
205  return dt_real;
206 }

212 template <class PST, class OPE, class VAR>
213 void TimeDiscretization<PST, OPE, VAR>::execute() {
214  // Initialization
215  this->initialize();
216  // Variable& var2 = variables_.getIVariable(0);
217  auto& var = variables_.get_variable("phi");
218  // Not necessary : deja dans Operator au constructeur
219  // this->oper_.initialize(var);
220  auto current_time = this->initial_time_;
221  auto unk = var.get_unknown();
222 
223  //=============================================
224  // loop over time-step
225  //=============================================
226  for (auto iter = 1; !this->last_step_; iter++) {
227  //------------
228  // Time-step
229  //------------
230  auto current_time_step = compute_real_time_step(current_time);
231  this->print_time_step_info(iter, current_time, this->time_step_, current_time_step);
232  //------------
233  // Solve
234  //------------
235  this->ode_solver_->Step(unk, current_time, current_time_step);
236  //-----------------
237  // Update solution
238  //-----------------
239  var.update(unk);
240 
241  //---------------------
242  // Extra calculations
243  //---------------------
244  // oper.SetParameters(unk);
245  if (this->compute_energies_) {
246  this->oper_.ComputeEnergies(std::make_tuple(iter, current_time_step, current_time), unk);
247  }
248  if (this->compute_error_) {
249  auto solution_func = this->analytical_solution_.get();
250 
251  this->oper_.ComputeError(std::make_tuple(iter, current_time_step, current_time), unk,
252  *solution_func);
253  }
254  //-------------------
255  // Visualization
256  //-------------------
257  if (this->pst_.need_to_be_saved(iter)) {
258  this->pst_.save_variables(this->variables_, iter, current_time);
259  }
260  }
261  if (this->compute_error_) {
262  this->pst_.export_csv("error.csv", this->oper_.get_l2_error());
263  }
264  if (this->compute_energies_) {
265  this->pst_.export_csv("energy_density.csv", this->oper_.get_energy_density());
266  this->pst_.export_csv("energy_interface.csv", this->oper_.get_energy_interface());
267  }
268  //=============================================
269  // end of loop over time-step
270  //=============================================
271 }
