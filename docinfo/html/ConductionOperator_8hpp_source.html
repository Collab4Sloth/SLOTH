122 void ConductionOperator::Mult(const mfem::Vector &u, mfem::Vector &du_dt) const {
123  // Compute:
124  // du_dt = M^{-1}*-Ku
125  // for du_dt, where K is linearized by using u from the previous timestep
126  // z = K.u
127  Kmat.Mult(u, z);
128  z.Neg(); // z = -z
129  // du_dt = M * K.u
130 
131  // for (int i = 0; i < ess_tdof_list.Size(); i++)
132  // z(ess_tdof_list[i]) = 0.0; // set Dirichlet condition by hand
133 
134  M_solver.Mult(z, du_dt);
135 }

137 void ConductionOperator::ImplicitSolve(const double dt, const mfem::Vector &u,
138  mfem::Vector &du_dt) {
139  // Solve the equation:
140  // du_dt = M^{-1}*[-K(u + dt*du_dt)]
141  // for du_dt, where K is linearized by using u from the previous timestep
142  if (!T) {
143  T = Add(1.0, Mmat, dt, Kmat);
144  current_dt = dt;
145  T_solver.SetOperator(*T);
146  }
147  MFEM_VERIFY(dt == current_dt, ""); // SDIRK methods use the same dt
148  Kmat.Mult(u, z);
149  z.Neg();
150 
151  T_solver.Mult(z, du_dt);
152  du_dt.SetSubVector(ess_tdof_list, 0.0);
153 }

155 void ConductionOperator::SetParameters(const mfem::Vector &u) {
156  mfem::GridFunction u_alpha_gf(&fespace);
157  u_alpha_gf.SetFromTrueDofs(u);
158  for (int i = 0; i < u_alpha_gf.Size(); i++) {
159  u_alpha_gf(i) = kappa + alpha * u_alpha_gf(i);
160  }
161 
162  delete K;
163  K = new mfem::BilinearForm(&fespace);
164 
165  mfem::GridFunctionCoefficient u_coeff(&u_alpha_gf);
166 
167  K->AddDomainIntegrator(new mfem::DiffusionIntegrator(u_coeff));
168  K->Assemble();
169  K->FormSystemMatrix(ess_tdof_list, Kmat);
170  delete T;
171  T = NULL; // re-compute T on the next ImplicitSolve
172 }
